<!DOCTYPE html>
<!--

Write a responsive web application that displays a textbox and a button.
If the screen is large, the textbox will have 9 columns and the button 3 columns. 
If the screen is smaller than that, they will both have 6 columns.
Type a text in the textbox and press the button.
When the button is pressed a sound with the frequency equal to the double of the ASCII code 
of each character should be played for half a second.

-->

<html lang="en">
    <head>
        <meta charset="UTF-8" />
        <meta name="viewport" content="width=device-width, initial-scale=1.0" />
        <title>Exam</title>
        <link href="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/css/bootstrap.min.css" rel="stylesheet" integrity="sha384-T3c6CoIi6uLrA9TneNEoa7RxnatzjcDSCmG1MXxSR1GAsXEV/Dwwykc2MPK8M2HN" crossorigin="anonymous" />
    </head>
    <body>
        <div class="container text-center">
            <div class="row">
                <div class="col-6 col-lg-9">
                    <input class="form-control" id="input" placeholder="Text" />
                </div>
                <div class="col-6 col-lg-3 d-grid">
                    <button type="button" class="btn btn-primary" id="play">Play</button>
                </div>
            </div>
        </div>
        <script src="https://cdn.jsdelivr.net/npm/bootstrap@5.3.2/dist/js/bootstrap.bundle.min.js" integrity="sha384-C6RzsynM9kWDrMNeT87bh95OGNyZPhcTNXj1NW7RuBCsyN/o0jlpcV8Qyq46cDfL" crossorigin="anonymous"></script>
        <script>
            document.getElementById('play').addEventListener('click', () => {
                let input = document.getElementById('input');
                let text = new String(input.value);
                let frequency = 0;
                for (let i = 0; i < text.length; i++) {
                    frequency += text.charCodeAt(i) * 2;
                }

                let ctx = new AudioContext();
                let oscillator = ctx.createOscillator();
                oscillator.type = 'sine';
                oscillator.frequency.setValueAtTime(frequency, ctx.currentTime);
                oscillator.connect(ctx.destination);
                oscillator.start();

                setTimeout(() => {
                    ctx.suspend();
                }, 500);
            });
        </script>
    </body>
</html>
